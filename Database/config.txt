#--------------------------------------------------------------------
# Ключ=Значение (пробелы учитываются)
# Разделителем считается только =
# Строка начинающаяся с # не учитывается
# Строки начинающиеся с @ считаются именем типа
# Некорректные строки игнорируются

#--------------------------------------------------------------------
# DEBUG - активация/отключение вывода сообщений определенных уровней
#       = 000 - отключить любые сообщения
#       = 100 - включить только критические
#       = 010 - включить только предупреждения
#       = 001 - включить только успешные
# Можно комбинировать несколько битов, например, 111 - включить все
#--------------------------------------------------------------------
# По умолчанию: DEBUG=111

DEBUG=111

#--------------------------------------------------------------------
# LOGS - настройка направления вывода логов СУБД
#      = cout     - вывод в стандартный поток вывода
#      = file.log - в остальных случах - имя файла вывода
#--------------------------------------------------------------------
# По умолчанию: LOGS=database.log

LOGS=database.log
#LOGS=cout

#--------------------------------------------------------------------
# SHOW_TIME - отображать время в логах (да/нет)
#           = True - выводить время
#           = False (или все остальные случаи)
#--------------------------------------------------------------------
# По умолчанию: SHOW_TIME=True

SHOW_TIME=True

#--------------------------------------------------------------------
# TIME_FORMAT - формат, в котором выводится время, описание:
#               http://www.cplusplus.com/reference/ctime/strftime/
#--------------------------------------------------------------------
# По умолчанию: TIME_FORMAT=%d.%m.%Y %I:%M:%S

TIME_FORMAT=%d.%m.%Y %I:%M:%S

#--------------------------------------------------------------------
# STD_DELIMS - стандартный разделитель значений в таблицах
#--------------------------------------------------------------------
# По умолчанию: STD_DELIMS=|

STD_DELIMS=|:

#--------------------------------------------------------------------
# SIG_CRIT - обозначение критического сообщения
#--------------------------------------------------------------------
# По умолчанию: SIG_CRIT=[!] (пробел на конце)

SIG_CRIT=[!] 

#--------------------------------------------------------------------
# SIG_WARN - обозначение предупреждющего сообщения
#--------------------------------------------------------------------
# По умолчанию: SIG_WARN=[-] (пробел на конце)

SIG_WARN=[-] 

#--------------------------------------------------------------------
# SIG_NORM - обозначение успешного сообщения
#--------------------------------------------------------------------
# По умолчанию: SIG_NORM=[+] (пробел на конце)

SIG_NORM=[+] 

#--------------------------------------------------------------------
# PRINT_PRECISION - точность вывода нецелых значений
#                 = 0 - строго не устанавливать, переменной длины
#                 = X - выводить строго X знаков после запятой
#--------------------------------------------------------------------
# По умолчанию: PRINT_PRECISION=0

PRINT_PRECISION=1

#--------------------------------------------------------------------
# @ИмяТипа=1 - int/unsigned
# @ИмяТипа=2 - double/float
# @ИмяТипа=3 - string/char*
#--------------------------------------------------------------------

@Integer=1
@Int32=1
@Size=1
@Unsigned=1
@Double=2
@Float=2
@Real=2
@String=3
@Char=3
@Char*=3
@Date=4
@DBDate=4

#--------------------------------------------------------------------
